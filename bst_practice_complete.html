Building Tree from Preorder + Inorder

// Build BST from Preorder and Inorder traversals
class TreeBuilder {
    int preIndex = 0;  // Current position in preorder array
    
    public Node buildFromPreIn(int[] preorder, int[] inorder) {
        preIndex = 0;
        return buildPreIn(preorder, inorder, 0, inorder.length - 1);
    }
    
    private Node buildPreIn(int[] pre, int[] in, int inStart, int inEnd) {
        // Base case: no more elements in this subtree
        if (inStart > inEnd) {
            return null;
        }
        
        // Step 1: Next element in preorder is the root
        int rootValue = pre[preIndex++];
        Node root = new Node(rootValue);
        
        // Step 2: Find root position in inorder
        int rootIndex = findIndex(in, inStart, inEnd, rootValue);
        
        // Step 3: Build left subtree (elements before root in inorder)
        root.left = buildPreIn(pre, in, inStart, rootIndex - 1);
        
        // Step 4: Build right subtree (elements after root in inorder)
        root.right = buildPreIn(pre, in, rootIndex + 1, inEnd);
        
        return root;
    }
    
    // Helper: Find value's index in array
    private int findIndex(int[] arr, int start, int end, int value) {
        for (int i = start; i <= end; i++) {
            if (arr[i] == value) return i;
        }
        return -1;
    }
}

// Example usage:
int[] preorder = {20, 10, 4, 8, 12, 14, 22};
int[] inorder = {4, 8, 10, 12, 14, 20, 22};

TreeBuilder builder = new TreeBuilder();
Node root = builder.buildFromPreIn(preorder, inorder);

Building Tree from Postorder + Inorder

// Build BST from Postorder and Inorder traversals
class TreeBuilder {
    int postIndex;  // Current position in postorder (starts from END)
    
    public Node buildFromPostIn(int[] postorder, int[] inorder) {
        postIndex = postorder.length - 1;  // Start from LAST element
        return buildPostIn(postorder, inorder, 0, inorder.length - 1);
    }
    
    private Node buildPostIn(int[] post, int[] in, int inStart, int inEnd) {
        if (inStart > inEnd) {
            return null;
        }
        
        // Step 1: LAST unprocessed element in postorder is root
        int rootValue = post[postIndex--];  // Decrement (going backwards)
        Node root = new Node(rootValue);
        
        // Step 2: Find root in inorder
        int rootIndex = findIndex(in, inStart, inEnd, rootValue);
        
        // Step 3: Build RIGHT subtree FIRST (postorder processes right before left)
        root.right = buildPostIn(post, in, rootIndex + 1, inEnd);
        
        // Step 4: Build LEFT subtree
        root.left = buildPostIn(post, in, inStart, rootIndex - 1);
        
        return root;
    }
    
    private int findIndex(int[] arr, int start, int end, int value) {
        for (int i = start; i <= end; i++) {
            if (arr[i] == value) return i;
        }
        return -1;
    }
}

// Example usage:
int[] postorder = {5, 15, 10, 25, 35, 30, 20};
int[] inorder = {5, 10, 15, 20, 25, 30, 35};

TreeBuilder builder = new TreeBuilder();
Node root = builder.buildFromPostIn(postorder, inorder);
Important Differences:
  • Preorder: Process left-to-right, root is FIRST, build left subtree then right
  • Postorder: Process right-to-left, root is LAST, build right subtree FIRST then left
  • Both: Use inorder to split left and right subtrees
